# 天梯赛 L2

### Code001

```c++
//测试链接:https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=994805073643683840&page=1
#include <bits/stdc++.h>
using namespace std;

const int N = 520;
const int INF = 0x7fffffff;
int G[N][N];
int pre[N];
int dis[N];
int road[N];
int cnt[N];
int cntsum[N];
bool vis[N];
int n, m, s, d;

void printPath(int v) {
    if (v == s) {
        cout << v;
        return;
    }
    printPath(pre[v]);
    cout << " " << v;
}

int main() {
    cin >> n >> m >> s >> d;
    fill(G[0], G[0] + N * N, INF);
    fill(dis, dis + N, INF);

    for(int i = 0; i < n; i++) {
        cin >> cnt[i];
    }

    int tempa, tempb, tempc;
    for(int i = 0; i < m; i++) {
        cin >> tempa >> tempb >> tempc;
        G[tempa][tempb] = tempc;
        G[tempb][tempa] = tempc;
    }

    dis[s] = 0;
    road[s] = 1;
    cntsum[s] = cnt[s];

    for(int i = 0; i < n; i++) {
        int u = -1, min = INF;
        for(int j = 0; j < n; j++) {
            if(vis[j] == false && dis[j] < min) {
                u = j;
                min = dis[j];
            }
        }
        if (u == -1) break;
        vis[u] = true;
        for(int v = 0; v < n; v++) {
            if (G[u][v] != INF && vis[v] == false) {
                if(dis[u] + G[u][v] < dis[v]){
                    dis[v] = dis[u] + G[u][v];
                    cntsum[v] = cntsum[u] + cnt[v];
                    road[v] = road[u];
                    pre[v] = u;
                } else if (dis[v] == dis[u] + G[u][v]) {
                    road[v] += road[u];
                    if (cntsum[v] < cntsum[u] + cnt[v]) {
                        cntsum[v] = cntsum[u] + cnt[v];
                        pre[v] = u;
                    }
                }
            }
        }
    }
    cout << road[d] << " " << cntsum[d] << endl;
    printPath(d);
    return 0;
}
```

### Code002

```c++
//测试链接:https://pintia.cn/problem-sets/994805046380707840/exam/problems/type/7?problemSetProblemId=994805072641245184&page=1
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn = 1e5;  // 最大节点数
struct Node {
    int address;  // 节点地址
    int key;      // 键值
    int next;     // 下一个节点地址
    int num;      // 记录数组下标位置，用于排序
} node[maxn];     // 节点数组

bool vis[maxn];   // 用于记录键值绝对值是否已经出现过

// 排序规则：按 num 值从小到大排序
bool cmp(Node a, Node b) {
    return a.num < b.num;
}

int main() {
    int head, n, a;
    scanf("%d%d", &head, &n);  // 输入链表头地址和节点总数

    int k1 = 0, k2 = 0;  // k1 记录无重复节点，k2 记录重复节点

    // 初始化每个节点的 num 为 2*maxn，确保位置分布正确
    for (int i = 0; i < maxn; i++) {
        node[i].num = 2 * maxn;
    }

    // 读取每个节点的信息
    for (int i = 0; i < n; i++) {
        scanf("%d", &a);  // 节点地址
        scanf("%d%d", &node[a].key, &node[a].next);  // 键值和下一个节点地址
        node[a].address = a;  // 存储节点地址
    }

    // 遍历链表，处理去重
    for (int i = head; i != -1; i = node[i].next) {
        if (!vis[abs(node[i].key)]) {  // 如果绝对值没有出现
            vis[abs(node[i].key)] = true;
            node[i].num = k1;  // 标记为无重复节点
            k1++;
        } else {  // 绝对值已经出现
            node[i].num = maxn + k2;  // 标记为重复节点
            k2++;
        }
    }

    // 按 num 对所有节点排序
    sort(node, node + maxn, cmp);

    int k = k1 + k2;  // 总共需要输出的节点数

    // 输出节点
    for (int i = 0; i < k; i++) {
        if (i != k1 - 1 && i != k - 1) {
            // 输出中间节点，下一节点地址需补全为 5 位
            printf("%05d %d %05d\n", node[i].address, node[i].key, node[i+1].address);
        } else {
            // 输出最后一个节点，下一节点地址为 -1
            printf("%05d %d -1\n", node[i].address, node[i].key);
        }
    }

    return 0;
}
```
